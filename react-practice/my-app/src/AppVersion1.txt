import react, {useState} from "react";
import reactDOM from "react-dom/client";
import "./App.css";
import ProfileCard from "./components/ProfileCard";
import Header from "./components/Header";
import Footer from "./components/Footer";
import Greeting from "./components/Greeting";
import Event from "./components/Event"
import SaySomething from "./components/SaySomething";

function SayHi(){
  const [message, setMessage] = useState("");
  const handleHi = () =>{
    setMessage("Hii Vedant");
  }
  return(
    <>
    <button onClick={handleHi}>Say Hii</button>
    <h1>{message}</h1>
    </>
  )
}

function SayBye(){
  const [message, setMessage] = useState("");
  const handleBye = () =>{
    setMessage("Bye Vedant");
  }

  return(
    <>
    <button onClick={handleBye}>Say Bye</button>
    <h1>{message}</h1>
    </>
  )
}


function App(){
  const name = "Vedant";
  const [count, setCount] =  useState(0);
  const [likes, setLikes] = useState([0,0,0]);

  const handleLike = (index)=>{
    const newLikes = [...likes];
    newLikes[index] += 1;
    setLikes(newLikes); 
  }
  return (<>
  <Header></Header>
  <Greeting name = "Everyone" ></Greeting>
  <Event></Event>
  <h1>This is a counter.</h1>
  <h3>Count: {count}</h3>
  <button onClick={()=>setCount(count +1)}>Increase</button>
  <button onClick={()=>setCount(count -1)}>Decrease</button>
    <SayHi></SayHi>
    <SayBye></SayBye>
    <SaySomething></SaySomething>
  <ProfileCard name = "Vedant Hippalgave" age = "20" tagline = "When nothing is for sure, anything can happen..."></ProfileCard>
  <h2>Like: {likes[0]}</h2>
  <button onClick={()=>handleLike(0)}>👍🏻</button>
  <ProfileCard name = "Bill Gates" age = "54" tagline = "I am the owner of Microsoft"></ProfileCard>
  <h2>Like: {likes[1]}</h2>
  <button onClick={()=>handleLike(1)}>👍🏻</button>
  <ProfileCard name = "Virat Kohli" age = "36" tagline = "I am the Best"></ProfileCard>
  <h2>Like: {likes[2]}</h2>
  <button onClick={()=>handleLike(2)}>👍🏻</button>
  <h2>Learning react feels like express but for UI</h2>
  <Footer></Footer>
  </>
  )
}

export default App;
